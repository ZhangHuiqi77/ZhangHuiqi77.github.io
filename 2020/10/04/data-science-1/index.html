# for loop
sum = 0
for number in range(11):
    sum = sum + number
print(sum)

# range function means 0:1:11, range(11) = range(0,11), 11 is not included, output = 55
# range(1,11,2) means [1,3,5,7,9], 1 is included but 11 is not included

# while loop, conditional loop, which also has advantage for the uncertain cycle numbers

sum = 0
number = 1
while number < 11:
    sum = sum + number
    number = number +1
print(sum)
